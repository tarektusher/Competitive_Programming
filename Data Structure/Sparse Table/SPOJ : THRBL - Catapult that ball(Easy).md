const long long maX =1e6+10;
const ll k=31;
ll sp[maX][k],ara[maX],lg[maX];
ll n,m;
void cal(){
    for(ll i=1;i<=n;i++)sp[i][0]=ara[i];
    for(ll j=1;j<k;j++){
         for(ll i=1;(i+(1<<j)-1)<=n;i++){
              sp[i][j]=max(sp[i][j-1],sp[i+(1<<(j-1))][j-1]);
         }
    }
}
ll get(ll l,ll r){
     ll j=lg[r-l+1];
     return max(sp[l][j],sp[r-(1<<j)][j]);
}
